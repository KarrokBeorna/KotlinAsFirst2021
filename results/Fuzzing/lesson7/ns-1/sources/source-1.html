25 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
26     val writer = File(outputName).bufferedWriter()
27     var currentLineLength = 0
28     fun append(word: String) {
29         if (currentLineLength > 0) {
30             if (word.length + currentLineLength >= lineLength) {
31                 writer.newLine()
32                 currentLineLength = 0
33             } else {
34                 writer.write(" ")
35                 currentLineLength++
36             }
37         }
38         writer.write(word)
39         currentLineLength += word.length
40     }
41     for (line in File(inputName).readLines()) {
42         if (line.isEmpty()) {
43             writer.newLine()
44             if (currentLineLength > 0) {
45                 writer.newLine()
46                 currentLineLength = 0
47             }
48             continue
49         }
50         for (word in line.split(Regex("\\s+"))) {
51             append(word)
52         }
53     }
54     writer.close()
55 }

292 fun markdownToHtmlSimple(inputName: String, outputName: String) {
293     val file = preprocessString(File(inputName).readText())
294 
295     var processed = Regex("\\*\\*(.*?)\\*\\*").replace(file) { m ->
296         "<b>" + m.value.replace("**", "") + "</b>"
297     }
298     processed = Regex("\\*(.*?)\\*").replace(processed) { m ->
299         "<i>" + m.value.replace("*", "") + "</i>"
300     }
301     processed = Regex("~~(.*?)~~").replace(processed) { m ->
302         "<s>" + m.value.replace("~~", "") + "</s>"
303     }
304 
305     val result = processed.split("_??_")
306 
307     val linesToWrite = mutableListOf("<p>")
308     var counterInParagraph = 0
309     var x = 0
310     for (line in result) {
311         if ((line.replace("\t", "") != "") && (line.replace(Regex("\\s"), "") != "")) {
312             linesToWrite.add(line.replace("\t", ""))
313             counterInParagraph += 1
314         } else {
315             if (counterInParagraph != 0 && x + 1 != result.size) {
316                 linesToWrite.add("</p><p>")
317                 counterInParagraph = 0
318             }
319         }
320         x += 1
321     }
322     if (counterInParagraph == 0 && processed.trim() != "" && x != 2) {
323         linesToWrite.removeLast()
324     }
325     linesToWrite.add("</p>")
326     File(outputName).writeText(createHTML(linesToWrite.joinToString(separator = "\n")))

428     val notClosedLists = mutableListOf<String>()
429     val linesToWrite: MutableList<String> = mutableListOf()
430     val dots = mapOf(
431         "*" to "ul",
432         "1" to "ol"
433     )
434     var before: Pair<Int, String> = -1 to ""
435     for (x in file.indices) {
436         val line = file[x]
437         var dot = line.trim().split(" ")[0]
438         val tabs = line.indexOf(dot)
439         val text = line.substring(tabs + dot.length + 1)
440         if (dot != "*") {
441             dot = "1"
442         }
443         if (tabs > before.first) {
444             val lastIndex = linesToWrite.lastIndex
445             if (linesToWrite.isNotEmpty() && linesToWrite[lastIndex].isNotEmpty())
446                 linesToWrite[lastIndex] =
447                     linesToWrite[lastIndex].substring(0, linesToWrite[lastIndex].length - 5)
448             notClosedLists.add("li")
449             linesToWrite.add("<${dots[dot]}>")
450             notClosedLists.add("${dots[dot]}")
451             linesToWrite.add("<li>${text}</li>")
452         } else if (tabs < before.first) {
453             linesToWrite.add("</${dots[before.second]}></li>")
454             notClosedLists.remove("li")
455             notClosedLists.remove("${dots[before.second]}")
456             linesToWrite.add("<li>${text}</li>")
457         } else if (before.second != dot) {
458             linesToWrite.add("</${dots[before.second]}>")
459             notClosedLists.remove("${dots[before.second]}")
460             linesToWrite.add("<li>${text}</li>")
461         } else {
462             linesToWrite.add("<li>${text}</li>")
463         }
464         before = tabs to dot
465     }
466     notClosedLists.removeFirst()
467     for (element in notClosedLists.reversed()) {
468         linesToWrite.add("</$element>")
469     }
470     File(outputName).writeText(createHTML("<p>" + linesToWrite.joinToString(separator = "\n") + "</p>"))
471 }
